import React from 'react';
import toastr from 'toastr';
import AppActions from '../actions/AppActions';
import Header from '../components/Navbar.jsx';
/**
 * @class ResetPassword
 * 
 * @extends {React.Component}
 * 
 * @return {void}
 */
class ResetPassword extends React.Component {
  /**
   * Creates an instance of ResetPassword.
   * 
   * @param {any} props
   * 
   * @memberof ResetPassword
   */
  constructor(props) {
    super(props);
    this.state = {
      email: '',
      message: ''
    };
  }
  /**
   * Creates an on change event method
   * 
   * @param {any} event
   * 
   * @memberof ResetPassword
   * 
   * @return {void}
   */
  onChange = (event)=> {
    this.setState({
      [event.target.name]: event.target.value
    });
  }
  /**
   * Creates an on submit event that fires a submit action.
   * @return {void}
   * @param {any} event
   * @memberof ResetPassword
   */
  onSubmit = (event) => {
    event.preventDefault();
    const resetEmail = {
      email: this.state.email
    };
    AppActions.resetPassword(resetEmail)
      .then(() => {
        this.props.history.push('/signin');
      })
  }
 
  /**
   * @memberof ResetPassword
   * 
   * @returns {void}
   */
  render() {
    return (
      <div>
        <Header />
        <form id="resetpassword" onSubmit={this.onSubmit}>
          <fieldset className="account-info">
            <label>
              Email Address
              <input value={this.state.email} onChange={this.onChange}
                type="email" name="email" required />
            </label>
          </fieldset>
          <button name="login" className="btn btn-primary btn-sm">
            Reset Password
         </button>
        </form>
      </div>
    );
  }
}
export default ResetPassword;
